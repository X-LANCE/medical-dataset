sql = Intersect(sql_unit, sql_unit)
        | Union(sql_unit, sql_unit)
        | Except(sql_unit, sql_unit)
        | Single(sql_unit)

sql_unit = Complete(select, from, cond, group_by, order_by)
        | NoWhere(select, from, group_by, order_by)
        | NoGroupBy(select, from, cond, order_by)
        | NoOrderBy(select, from, cond, group_by)
        | OnlyWhere(select, from, cond)
        | OnlyGroupBy(select, from, group_by)
        | OnlyOrderBy(select, from, order_by)
        | Simple(select, from)

select = SelectOne(val_unit)
        | SelectTwo(val_unit, val_unit)
        | SelectThree(val_unit, val_unit, val_unit)
        | SelectFour(val_unit, val_unit, val_unit, val_unit)
        | SelectFive(val_unit, val_unit, val_unit, val_unit, val_unit)
        | SelectSix(val_unit, val_unit, val_unit, val_unit, val_unit, val_unit)

from = FromOneTable(tab)
        | FromTwoTable(tab, tab)
        | FromThreeTable(tab, tab, tab)
        | FromFourTable(tab, tab, tab, tab)
        | FromFiveTable(tab, tab, tab, tab, tab)
        | FromSixTable(tab, tab, tab, tab, tab, tab)
        | FromSevenTable(tab, tab, tab, tab, tab, tab, tab)
        | FromSQL(sql)

group_by = OneNoHaving(col_unit)
        | TwoNoHaving(col_unit, col_unit)
        | ThreeNoHaving(col_unit, col_unit, col_unit)
        | OneHaving(col_unit, cond)
        | TwoHaving(col_unit, col_unit, cond)
        | ThreeHaving(col_unit, col_unit, col_unit, cond)

order_by = OneAsc(val_unit)
        | OneDesc(val_unit)
        | OneAscLimit(val_unit)
        | OneDescLimit(val_unit)
        | TwoAsc(val_unit, val_unit)
        | TwoDesc(val_unit, val_unit)
        | TwoAscLimit(val_unit, val_unit)
        | TwoDescLimit(val_unit, val_unit)
        | ThreeAsc(val_unit, val_unit, val_unit)
        | ThreeDesc(val_unit, val_unit, val_unit)
        | ThreeAscLimit(val_unit, val_unit, val_unit)
        | ThreeDescLimit(val_unit, val_unit, val_unit)

cond = And(cond, cond)
        | Or(cond, cond)
        | Between(val_unit)
        | Eq(val_unit)
        | Gt(val_unit)
        | Lt(val_unit)
        | Ge(val_unit)
        | Le(val_unit)
        | Neq(val_unit)
        | Like(val_unit)
        | NotLike(val_unit)
        | BetweenSQL(val_unit, sql)
        | EqSQL(val_unit, sql)
        | GtSQL(val_unit, sql)
        | LtSQL(val_unit, sql)
        | GeSQL(val_unit, sql)
        | LeSQL(val_unit, sql)
        | NeqSQL(val_unit, sql)
        | InSQL(val_unit, sql)
        | NotInSQL(val_unit, sql)

val_unit = Unary(col_unit)
        | Minus(col_unit, col_unit)
        | Plus(col_unit, col_unit)
        | Times(col_unit, col_unit)
        | Divide(col_unit, col_unit)

col_unit = None(col)
        | Max(col)
        | Min(col)
        | Count(col)
        | Sum(col)
        | Avg(col)
        | Distinct(col)
        | CountDistinct(col)
